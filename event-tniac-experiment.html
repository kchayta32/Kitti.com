% I: image (RGB or gray)
% bbox_xywh: [x y w h] in pixels

  % --- crop ROI + padding ---
  pad = 0.10; % 10% padding
  x = bbox_xywh(1); y = bbox_xywh(2);
  w = bbox_xywh(3); h = bbox_xywh(4);

  % --- preprocess ---
  G = rgb2gray(ROI);
  G = imgaussfilt(G, 1.0);
  E = edge(G, "Canny");

  % --- Hough line detection ---

  P = houghpeaks(Hh, 8);
