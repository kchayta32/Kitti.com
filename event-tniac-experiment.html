% I: image (RGB or gray)
% bbox_xywh: [x y w h] in pixels

  % --- crop ROI + padding ---
  pad = 0.10; % 10% padding
  x = bbox_xywh(1); y = bbox_xywh(2);
  w = bbox_xywh(3); h = bbox_xywh(4);

  G = imgaussfilt(G, 1.0);
